Redis ​

下载地址：https://redis.io/download/

全部版本：https://download.redis.io/releases/

最新稳定版： https://download.redis.io/redis-stable.tar.gz

源码编译安装 ​

bash
#!/bin/bash

redis_version=redis-7.2.0
redis_dir=/apps/redis
cpus=`nproc`
PASSWORD=123456

color() {
    echo -e "\e[32m$*\e[0m"
}
color 安装依赖...
dnf install -y make gcc > /dev/null



cd /opt
if [ ! -e ${redis_version}.tar.gz ]; then
	wget  https://download.redis.io/releases/${redis_version}.tar.gz
fi

tar xf ${redis_version}.tar.gz -C /usr/local/src
cd /usr/local/src/$redis_version
make -j $cpus && make install || { color error; exit; }
#ln -sv ${redis_dir}/bin/redis-* /usr/local/bin

# 默认安装位置 /usr/local/bin

mkdir -p $redis_dir/{log,data,run}
id redis &> /dev/null || useradd -s /sbin/nologin -r redis
chown -R redis:redis $redis_dir


cp redis.conf $redis_dir

# 修改配置
# daemonize yes
sed -ri -e "s/(^daemonize ).*/\1yes/" \
        -e "s/(^bind ).*/\10.0.0.0/" \
        -e "/^pidfile/c pidfile ${redis_dir}\/redis_6379.pid" \
        -e "/^logfile/c logfile ${redis_dir}\/redis_6379.log" \
        -e "/# requirepass/a requirepass $PASSWORD" \
        $redis_dir/redis.conf



# 消除 memory overcommit警告，可选
echo "vm.overcommit_memory = 1" >> /etc/sysctl.conf
sysctl vm.overcommit_memory=1

# 启动服务端
# /user/local/bin/redis-server /myredis/redis.conf
color 启动redis_server...
redis-server $redis_dir/redis.conf
sleep 2

# 连接
# 或者进入验证 auth password
# redis-cli -a password
#redis-cli -a $PASSWORD INFO Server
redis-cli -a $PASSWORD ping && color 安装完成
