sh
domain=example.com
email=zaiboluo.gmail.com


echo install bind
dnf install -y -q bind bind-utils

# 1 更改dns主配置文件
sed -i 's/localhost;/any;/' /etc/named.conf
sed -i 's/127.0.0.1/any/' /etc/named.conf
# 配置检查
# named-checkconf

# 2 创建区域记录文件
cat << EOF > /var/named/${domain}.zone
\$TTL 1D
@   IN  SOA    @    $email(
               1       ; Serial
               1D      ; Refresh
               10M     ; Retry
               1D      ; Expire
               1H      ; Minium TTL
               )
        NS     @

        A      10.0.0.10

;www   IN      A      10.0.0.10
www     A      10.0.0.10

; *     A      10.0.0.10
; “*”代表所有，就是泛域名解析
EOF

# 更改权限和组
chmod 640 /var/named/${domain}.zone
chgrp named /var/named/${domain}.zone

# 3 加入配置
if [ `grep -c $domain  /etc/named.rfc1912.zones` -eq 0 ]; then

    cat << EOF >> /etc/named.rfc1912.zones
zone "${domain}" IN {
    type master;
    file "${domain}.zone";
};
EOF
else
    echo $domain 已添加
fi

if [[ `systemctl is-active named` == 'active' ]]; then
    rndc reload
else
    systemctl enable --now named
fi

echo success
